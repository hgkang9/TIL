def Find(x):
    global p
    if x==P[x] or x in p:p+=[x];
    else:p+=[x];return Find(P[x])
T=int(input())
for n in range(T):
    L=[int(x)for x in input().split()]
    L1=[int(x)for x in input().split()]
    P=[0]+[1]*L[0]
    for i in range(0,len(L1),2):
        P[L1[i+1]]=L1[i]
    p=[]
    Find(L[2])
    Find(L[3])
    cnt=1
    R=[p[-1]]
    while R:
        a=R.pop(0)
        for i in range(0,len(L1),2):
            if L1[i]==a:R.append(L1[i+1]);cnt+=1
    print('#%d %d %d'%(n+1,p[-1],cnt))

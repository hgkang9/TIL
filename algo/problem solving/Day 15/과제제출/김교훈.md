### 예산 관리 (재귀)

``` python
B = int(input())
n = int(input())
data = list(map(int,input().split()))
max_sum=0
visited=[0]*n
def DFS(index, this_sum):
    global data
    global B
    global n
    global max_sum
    global visited
    if(this_sum>B):
        return
    if(index>n):
        return
    if(this_sum>max_sum):
        max_sum=this_sum
    for i in range(0,len(data)):
        if(visited[i]==0):
            visited[i] = 1
            DFS(index+1,this_sum+data[i])
            visited[i]=0
DFS(0,0)
print(max_sum)
```

``` python
    temp=[]
    # print(i)

    this_sum=sum(temp)

    if(this_sum>max_sum):

            max_sum = this_sum
print(max_sum)
```

``` python
import copy
data=list(map(int,raw_input().split()))

    if -1<Y<data[0] and -1<X<data[1]:

        return False

def this_air_state(y,x):
    global my_map
    global dy, dx
    blocked=0

    for dir in range(0,4):
        now_y = copy.deepcopy(y)
        now_x = copy.deepcopy(x)
        while(1):
            if ( is_safe(now_y+dy[dir],now_x+dx[dir]) == True ):
                if my_map[now_y+dy[dir]][now_x+dx[dir]] == 1:
                    blocked=blocked+1
                    break
                elif my_map[now_y + dy[dir]][now_x + dx[dir]] == 2:
                    break
                else:
                    now_y=now_y+dy[dir]
                    now_x=now_x+dx[dir]
            else:
                break
    if blocked==4:
        return True
    else:
        return False

def air_upload():
    global my_map

        for xx in range(0,data[1]):
```

####  사칙연산

``` python
T=10
for t in range(1,T+1):
    VALUE=0
    PARENT=1
    LC=2
    RC=3

    N=int(input())
    TREE=[]
    for i in range(0,N+1):
        line=[0]*4
        TREE.append(line)

    for i in range(0,N):
        data = list(input().split())

        if(len(data)>2):
            TREE[int(data[0])][VALUE] = data[1]
            TREE[int(data[0])][LC] = int(data[2])
            TREE[int(data[0])][RC] = int(data[3])
            TREE[int(data[2])][PARENT] = int(data[0])
            TREE[int(data[3])][PARENT] = int(data[0])
        else:
            TREE[int(data[0])][VALUE]=data[1]

    # print(TREE)
    i=len(TREE)-1

    while(i!=1):
        if TREE[TREE[i][PARENT]][VALUE] == "+":
            this_answer = int(TREE[i-1][VALUE]) + int(TREE[i][VALUE])
        if TREE[TREE[i][PARENT]][VALUE] == "-":
            this_answer = int(TREE[i-1][VALUE]) - int(TREE[i][VALUE])
        if TREE[TREE[i][PARENT]][VALUE] == "*":
            this_answer = int(TREE[i-1][VALUE]) * int(TREE[i][VALUE])
        if TREE[TREE[i][PARENT]][VALUE] == "/":
            this_answer = int(TREE[i-1][VALUE]) / int(TREE[i][VALUE])
        TREE[TREE[i][PARENT]][VALUE] = int(this_answer)
        i=i-2

    print("#{} {}" .format(t, TREE[1][VALUE]))
```

``` python
for t in range(1,T+1):

        pascal_line.append(pascal[i][len(pascal[i])-1])
        pascal.append(pascal_line)

    print("#{}".format(t))
    for i in pascal:
```

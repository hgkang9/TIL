T = int(input())

for tc in range(1, T+1):
    stack = []
    data = list(map(str, input().split()))

            stack.pop(-1)
            stack.pop(-1)
            stack.append(int(res))
        elif len(stack)>=2 and data[i] == '*':
            res = int(stack[-1]) * int(stack[-2])
            stack.pop(-1)
            stack.pop(-1)
            stack.append(int(res))
        elif len(stack)>=2 and data[i] == '/':
            res = int(stack[-2]) / int(stack[-1])
            stack.pop(-1)
            stack.pop(-1)
            stack.append(int(res))
